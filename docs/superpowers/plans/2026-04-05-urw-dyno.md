# URW Dyno Page Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a dyno experience and action menu to `/race` — after picking a car, choose between Drag Race, Dyno, or Cruise. The dyno shows the car on animated rollers with URW branding, engine sounds, and spec reveal.

**Architecture:** All new phases live in the existing `app/race/page.tsx` component, following the same `phase` state pattern. The dyno room background is a one-time Imagen generation stored in Supabase. Car direction is handled via a `pixel_art_flipped` boolean column and CSS/canvas transforms.

**Tech Stack:** React, Canvas API, Web Audio (existing `lib/race-audio.ts`), Imagen API (existing `lib/generate-pixel-art.ts`), Supabase

- [ ] **Step 2: Create the API endpoint**

Create `app/api/registrations/pixel-art/generate-dyno-room/route.ts`:

```typescript
import { NextResponse } from "next/server";
import { createServerClient } from "@/lib/supabase-server";
import { generateImage } from "@/lib/generate-pixel-art";

export async function POST() {
  try {
    const supabase = createServerClient();

    const prompt =
      `8-bit retro pixel art side view of an automotive dyno room interior. ` +
      `Clean white walls, polished gray concrete floor. A chassis dynamometer (roller dyno) is embedded in the floor ` +
      `with two large metal rollers visible. On the left side, a dyno computer kiosk/control station ` +
      `with a monitor showing a graph display. A technician in a dark uniform stands next to the kiosk ` +
      `holding a clipboard. On the back wall, the text "URW AUTOMOTIVE" is painted in large bold black ` +
      `letters. Fluorescent ceiling lights. An open garage door on the right side showing daylight outside. ` +
      `Style like a 1990s DOS racing game. Sharp pixels, no anti-aliasing, authentic retro video game aesthetic. ` +
      `Side-on view, no perspective distortion. No car on the dyno — the rollers are empty.`;

    const buffer = await generateImage(prompt);

    await supabase.storage.createBucket("pixel-art", { public: true, allowedMimeTypes: ["image/png"] });

    const fileName = "urw-dyno-room.png";
    await supabase.storage.from("pixel-art").upload(fileName, buffer, { contentType: "image/png", upsert: true });

    const url = `${supabase.storage.from("pixel-art").getPublicUrl(fileName).data.publicUrl}?v=${Date.now()}`;

    return NextResponse.json({ url });
  } catch (err) {
    console.error("Dyno room generation error:", err);
    return NextResponse.json({ error: "Failed to generate dyno room" }, { status: 500 });
  }
}
```

```typescript
    const bgImg = new Image();
    bgImg.crossOrigin = "anonymous";
    bgImg.src = DYNO_ROOM_URL;
    dynoBgRef.current = bgImg;

    setDynoPhase("pulling");
    setDynoStartTime(performance.now());
    startEngine();

    const redline = playerCar?.redline || 6500;
    const PULL_DURATION = 6000; // 6 seconds to reach redline
    const PEAK_DURATION = 1000; // hold at redline 1 second
    const particles: { x: number; y: number; vx: number; vy: number; life: number; color: string }[] = [];

    const animate = (now: number) => {
      const canvas = dynoCanvasRef.current;
      if (!canvas) return;
      const ctx = canvas.getContext("2d");
      if (!ctx) return;

      const W = canvas.width;
      const H = canvas.height;
      const elapsed = now - (dynoStartTime || now);

      // Calculate RPM
      let rpm: number;
      let phase: "pulling" | "peak" | "cooldown" | "done";

      if (elapsed < PULL_DURATION) {
        // Ease-in-out ramp
        const t = elapsed / PULL_DURATION;
        const eased = t < 0.5 ? 2 * t * t : 1 - Math.pow(-2 * t + 2, 2) / 2;
        rpm = 800 + (redline - 800) * eased;
        phase = "pulling";
      } else if (elapsed < PULL_DURATION + PEAK_DURATION) {
        rpm = redline;
        phase = "peak";
      } else if (elapsed < PULL_DURATION + PEAK_DURATION + 500) {
        const coolT = (elapsed - PULL_DURATION - PEAK_DURATION) / 500;
        rpm = redline * (1 - coolT);
        phase = "cooldown";
      } else {
        rpm = 0;
        phase = "done";
      }

      setDynoRpm(Math.round(rpm));
      if (phase !== "done") updateEngine(rpm, rpm / redline * 100);

      // Clear
      ctx.fillStyle = "#0d0d1a";
      ctx.fillRect(0, 0, W, H);

      // Draw background image
      if (dynoBgRef.current?.complete) {
        ctx.globalAlpha = 0.7;
        ctx.drawImage(dynoBgRef.current, 0, 0, W, H);
        ctx.globalAlpha = 1;
      }

      // Dyno rollers
      const rollerY = H * 0.82;
      const rollerR = 18;
      const rollerSpacing = W * 0.18;
      const rollerCX = W * 0.52;
      const rollerAngle = (elapsed * rpm / 50000) % (Math.PI * 2);

      for (const dx of [-rollerSpacing / 2, rollerSpacing / 2]) {
        const cx = rollerCX + dx;
        ctx.fillStyle = "#444";
        ctx.beginPath();
        ctx.arc(cx, rollerY, rollerR, 0, Math.PI * 2);
        ctx.fill();
        // Spokes
        ctx.strokeStyle = "#666";
        ctx.lineWidth = 2;
        for (let a = 0; a < 4; a++) {
          const angle = rollerAngle + (a * Math.PI / 2);
          ctx.beginPath();
          ctx.moveTo(cx + Math.cos(angle) * rollerR * 0.3, rollerY + Math.sin(angle) * rollerR * 0.3);
          ctx.lineTo(cx + Math.cos(angle) * rollerR * 0.9, rollerY + Math.sin(angle) * rollerR * 0.9);
          ctx.stroke();
        }
      }

      // Car sprite
      if (dynoImgRef.current?.complete) {
        const carW = W * 0.4;
        const carH = carW * (dynoImgRef.current.height / dynoImgRef.current.width);
        const carX = rollerCX - carW / 2;
        const vibFreq = 8 + (rpm / redline) * 17;
        const vibAmp = 1 + (rpm / redline) * 4;
        const vibY = Math.sin(elapsed * vibFreq / 100) * vibAmp;
        const carY = rollerY - carH + rollerR * 0.3 + vibY;

        ctx.save();
        if (playerCar?.flipped) {
          ctx.translate(carX + carW, carY);
          ctx.scale(-1, 1);
          ctx.drawImage(dynoImgRef.current, 0, 0, carW, carH);
        } else {
          ctx.drawImage(dynoImgRef.current, carX, carY, carW, carH);
        }
        ctx.restore();

        // Exhaust particles
        const exhaustX = playerCar?.flipped ? carX + carW + 5 : carX - 5;
        const exhaustDir = playerCar?.flipped ? 1 : -1;
        const spawnRate = 0.3 + (rpm / redline) * 2.7;
        if (phase !== "done" && Math.random() < spawnRate / 60) {

            y: rollerY - carH * 0.3 + vibY,

          ctx.beginPath();
          ctx.arc(p.x, p.y, 2 + (1 - p.life / 30) * 3, 0, Math.PI * 2);

          ctx.globalAlpha = 1;

      const gaugeY = H - gaugeH - 4;

      ctx.fillStyle = "#222";
      ctx.fillRect(10, gaugeY, gaugeW, gaugeH);
      const gaugeColor = gaugeFill < 0.6 ? "#00cc00" : gaugeFill < 0.85 ? "#ffcc00" : "#ff0000";
      ctx.fillStyle = gaugeColor;
      ctx.fillRect(10, gaugeY, gaugeW * gaugeFill, gaugeH);

        setDynoPhase("results");

        const hp = playerCar!.hp;
        const wt = playerCar!.weight;
        const et = quarterMileET(hp, wt, playerCar!.year, playerCar!.trans, playerCar!.gears);
        const trap = trapSpeedMPH(hp, wt, playerCar!.year, playerCar!.topSpeed);
        setDynoResults([
          { label: "HORSEPOWER", value: `${hp} HP`, revealed: false },
          { label: "WEIGHT", value: `${wt.toLocaleString()} LBS`, revealed: false },
          { label: "POWER/WEIGHT", value: `${playerCar!.pwr} HP/TON`, revealed: false },
          { label: "1/4 MILE", value: `${et.toFixed(1)}s`, revealed: false },

          ctx.drawImage(dynoImgRef.current, carX, carY, carW, carH);
        }
        ctx.restore();

        const exhaustX = playerCar?.flipped ? carX + carW + 5 : carX - 5;
        const exhaustDir = playerCar?.flipped ? 1 : -1;
        const spawnRate = 0.3 + (rpm / redline) * 2.7;
        if (currentPhase !== "done" && Math.random() < spawnRate / 60) {
          const isHot = rpm > redline * 0.8 && Math.random() < 0.3;
          particles.push({
            x: exhaustX,
            y: rollerY - carH * 0.3 + vibY,
            vx: exhaustDir * (1 + Math.random() * 2),
            vy: -(0.5 + Math.random()),
            life: 30,
            color: isHot ? "#ff6600" : "#888888",
          });
        }

        for (let i = particles.length - 1; i >= 0; i--) {
          const p = particles[i];
          p.x += p.vx;
          p.y += p.vy;
          p.life--;
          if (p.life <= 0) { particles.splice(i, 1); continue; }
          ctx.globalAlpha = p.life / 30;
          ctx.fillStyle = p.color;
          ctx.beginPath();
          ctx.arc(p.x, p.y, 2 + (1 - p.life / 30) * 3, 0, Math.PI * 2);
          ctx.fill();
          ctx.globalAlpha = 1;
        }
      }

      const gaugeH = 8;
      const gaugeY = H - gaugeH - 4;
      const gaugeW = W - 20;
      const gaugeFill = rpm / redline;
      ctx.fillStyle = "#222";
      ctx.fillRect(10, gaugeY, gaugeW, gaugeH);
      const gaugeColor = gaugeFill < 0.6 ? "#00cc00" : gaugeFill < 0.85 ? "#ffcc00" : "#ff0000";
      ctx.fillStyle = gaugeColor;
      ctx.fillRect(10, gaugeY, gaugeW * gaugeFill, gaugeH);

      if (currentPhase === "done") {
        stopEngine();
        setDynoPhase("results");
        const hp = playerCar!.hp;
        const wt = playerCar!.weight;
        const et = quarterMileET(hp, wt, playerCar!.year, playerCar!.trans, playerCar!.gears);
        const trap = trapSpeedMPH(hp, wt, playerCar!.year, playerCar!.topSpeed);
        setDynoResults([
          { label: "HORSEPOWER", value: `${hp} HP`, revealed: false },
          { label: "WEIGHT", value: `${wt.toLocaleString()} LBS`, revealed: false },
          { label: "POWER/WEIGHT", value: `${playerCar!.pwr} HP/TON`, revealed: false },
          { label: "1/4 MILE", value: `${et.toFixed(1)}s`, revealed: false },
          { label: "TRAP SPEED", value: `${Math.round(trap)} MPH`, revealed: false },
        ]);
        return;
      }

      dynoAnimRef.current = requestAnimationFrame(animateWithStart);
    };

    dynoAnimRef.current = requestAnimationFrame(animateWithStart);
  }, [playerCar]);
```

```typescript
function DynoResults({ results }: { results: { label: string; value: string; revealed: boolean }[]; onRevealComplete: () => void }) {
  const [revealedCount, setRevealedCount] = useState(0);

  useEffect(() => {
    if (revealedCount >= results.length) return;
    const timer = setTimeout(() => {
      // Play click sound
      try {
        const actx = new (window.AudioContext || (window as unknown as { webkitAudioContext: typeof AudioContext }).webkitAudioContext)();
        const osc = actx.createOscillator();
        const gain = actx.createGain();
        osc.connect(gain);
        gain.connect(actx.destination);
        osc.frequency.value = 80;
        osc.type = "square";
        gain.gain.value = 0.15;
        osc.start();
        osc.stop(actx.currentTime + 0.05);
      } catch {}
      setRevealedCount((c) => c + 1);
    }, 400);
    return () => clearTimeout(timer);
  }, [revealedCount, results.length]);

  return (
    <div style={{ display: "grid", gridTemplateColumns: "1fr 1fr", gap: "0.75rem", maxWidth: "400px", margin: "1rem auto" }}>
      {results.map((r, i) => (
        <div
          key={r.label}
          style={{
            padding: "0.75rem",
            background: i < revealedCount ? "rgba(255,215,0,0.08)" : "transparent",
            border: `1px solid ${i < revealedCount ? C.goldDark : C.border}`,
            opacity: i < revealedCount ? 1 : 0.2,
            transition: "all 0.3s ease",
            gridColumn: i === results.length - 1 && results.length % 2 !== 0 ? "1 / -1" : undefined,
          }}
        >
          <div style={{ fontFamily: FONT, fontSize: "0.6rem", color: C.midGray, marginBottom: "0.3rem" }}>{r.label}</div>
          <div style={{ fontFamily: FONT, fontSize: "1rem", color: C.gold }}>
            {i < revealedCount ? r.value : "---"}
          </div>
        </div>
      ))}
    </div>
  );
}
```
